    学数据分析,实现高薪就业捷径
    你想学习数据分析吗？
    一直希望sas可以有个调试器，可以暂停程序，查看宏变量赋值情况，数据集运行情况，于是自制了一个简陋的debugger，用于暂停程序，查看宏变量赋值情况（work.macro），欢迎大神拍砖，欢迎改善（觉得好用，拿走，记得给好评，呵呵）
    1. %macro Sys_Debug(debuginfo=SAS Debuging);
    2. /*options nonotes nomlogic nomprint nosymbolgen nosource nosource2;*/
    3. /*宏变量信息保存在work.macro*/
    4. proc sql noprint;
    5. create table macro as select *from DICTIONARY.macros
    6. where scope not contains ("AUTOMATIC");
    7. quit;
    8. data _null_;
    9.    window start
    10.           #9   @26 "WELCOME TO THE SAS SYSTEM" color=black
    11.                   #12 @26 "&debuginfo"
    12.           #15 @26 "Press ENTER to continue";
    13.     display start;
    14.     stop;
    15. run;
    16. %mend;
    复制代码
    给个例子吧，这个debug主要用于调试宏程序
    1. %macro step1;
    2. %let a=1;
    3. data a;
    4. do i=1 to 3;
    5. output;
    6. end;
    7. run;
    8. /*查看work.macro,检查宏变量j的信息，数据集a的信息，打开了数据集一定要关闭了才能按enter继续运行程序*/
    9. %do j=1 %to 3;
    10. data a;
    11. do i=1 to &j;
    12. output;
    13. end;
    14. run;
    15. %sys_debug;
    16. %end;
    17. %mend;
    18. %macro step2;
    19. %let b=1;
    20. %mend;
    21. %macro main;
    22. %step1;
    23. %step2;
    24. %mend;
    25. %main;
